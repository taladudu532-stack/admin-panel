  <script type="module">
    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.5.2/firebase-app.js";
    import { getAuth, signInWithEmailAndPassword, signOut } from "https://www.gstatic.com/firebasejs/10.5.2/firebase-auth.js";
    import { getFirestore, doc, getDoc } from "https://www.gstatic.com/firebasejs/10.5.2/firebase-firestore.js";

    const firebaseConfig = {
      apiKey: "YOUR_API_KEY",
      authDomain: "talabet-89fce.firebaseapp.com",
      projectId: "talabet-89fce",
      storageBucket: "talabet-89fce.appspot.com",
      messagingSenderId: "YOUR_SENDER_ID",
      appId: "YOUR_APP_ID"
    };

    const app = initializeApp(firebaseConfig);
    const auth = getAuth(app);
    const db = getFirestore(app);

    window.login = async function () {
      const email = prompt("أدخل البريد الإلكتروني");
      const password = prompt("أدخل كلمة المرور");
      try {
        const user = await signInWithEmailAndPassword(auth, email, password);
        document.getElementById("user-info").innerText = `مرحباً، ${user.user.email}`;
      } catch (err) {
        alert("فشل تسجيل الدخول");
      }
    };

    window.logout = async function () {
      try {
        await signOut(auth);
        document.getElementById("user-info").innerText = "تم تسجيل الخروج";
      } catch (err) {
        alert("فشل تسجيل الخروج");
      }
    };

    window.showBalance = async function (userId) {
      try {
        const userRef = doc(db, "users", userId);
        const snapshot = await getDoc(userRef);
        if (snapshot.exists()) {
          alert(`رصيدك: ${snapshot.data().coins} عملة`);
        } else {
          alert("المستخدم غير موجود");
        }
      } catch (err) {
        alert("تعذر جلب الرصيد");
      }
    };

    window.loadSportsGames = function () {
      alert("🏆 سيتم عرض الألعاب الرياضية هنا");
    };

    window.loadCasinoGames = function () {
      alert("🎰 سيتم عرض ألعاب الكازينو هنا");
    };

    window.loadClassicGames = function () {
      alert("🎮 سيتم عرض الألعاب الكلاسيكية هنا");
    };
