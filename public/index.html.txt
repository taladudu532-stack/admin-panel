  <script type="module">
    // Import Firebase modules
    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.5.2/firebase-app.js";
    import { 
      getAuth, 
      signInWithEmailAndPassword, 
      signOut, 
      onAuthStateChanged 
    } from "https://www.gstatic.com/firebasejs/10.5.2/firebase-auth.js";
    import { 
      getFirestore, 
      doc, 
      getDoc 
    } from "https://www.gstatic.com/firebasejs/10.5.2/firebase-firestore.js";

    // Firebase configuration - replace with your actual config
    const firebaseConfig = {
      apiKey: "YOUR_API_KEY",
      authDomain: "talabet-89fce.firebaseapp.com",
      projectId: "talabet-89fce",
      storageBucket: "talabet-89fce.appspot.com",
      messagingSenderId: "YOUR_SENDER_ID",
      appId: "YOUR_APP_ID"
    };

    // Initialize Firebase
    const app = initializeApp(firebaseConfig);
    const auth = getAuth(app);
    const db = getFirestore(app);

    // DOM elements
    const authContainer = document.getElementById('auth-container');
    const dashboard = document.getElementById('dashboard');
    const userEmail = document.getElementById('user-email');
    const userBalance = document.getElementById('user-balance');
    const notification = document.getElementById('notification');

    // Show notification function
    function showNotification(message, isError = false) {
      const icon = notification.querySelector('.notification-icon');
      const msg = notification.querySelector('.notification-message');
      
      msg.textContent = message;
      icon.innerHTML = isError ? '<i class="fas fa-exclamation-circle"></i>' : '<i class="fas fa-check-circle"></i>';
      notification.classList.toggle('error', isError);
      notification.classList.add('show');
      
      setTimeout(() => {
        notification.classList.remove('show');
      }, 3000);
    }

    // Check auth state
    onAuthStateChanged(auth, (user) => {
      if (user) {
        // User is signed in
        authContainer.style.display = 'none';
        dashboard.style.display = 'block';
        userEmail.textContent = user.email;
        loadUserData(user.uid);
      } else {
        // User is signed out
        authContainer.style.display = 'block';
        dashboard.style.display = 'none';
      }
    });

    // Login function
    window.login = async function () {
      const email = document.getElementById('email').value;
      const password = document.getElementById('password').value;
      
      if (!email || !password) {
        showNotification('يرجى ملء جميع الحقول', true);
        return;
      }
      
      try {
        await signInWithEmailAndPassword(auth, email, password);
        showNotification('تم تسجيل الدخول بنجاح');
      } catch (err) {
        console.error(err);
        showNotification('فشل تسجيل الدخول: ' + err.message, true);
      }
    };

    // Logout function
    window.logout = async function () {
      try {
        await signOut(auth);
        showNotification('تم تسجيل الخروج بنجاح');
      } catch (err) {
        console.error(err);
        showNotification('فشل تسجيل الخروج', true);
      }
    };

    // Load user data
    async function loadUserData(userId) {
      try {
        const userRef = doc(db, "users", userId);
        const snapshot = await getDoc(userRef);
        
        if (snapshot.exists()) {
          userBalance.textContent = snapshot.data().coins || 0;
        }
      } catch (err) {
        console.error("Error loading user data:", err);
      }
    }

    // Show balance function
    window.showBalance = async function () {
      const user = auth.currentUser;
      if (!user) return;
      
      try {
        const userRef = doc(db, "users", user.uid);
        const snapshot = await getDoc(userRef);
        
        if (snapshot.exists()) {
          showNotification(`رصيدك الحالي: ${snapshot.data().coins} عملة`);
        }
      } catch (err) {
        console.error(err);
        showNotification('تعذر جلب الرصيد', true);
      }
    };

    // Game loading functions
    window.loadSportsGames = function () {
      showNotification('سيتم تحميل الألعاب الرياضية قريباً');
    };

    window.loadCasinoGames = function () {
      showNotification('سيتم تحميل ألعاب الكازينو قريباً');
    };

    window.loadClassicGames = function () {
      showNotification('سيتم تحميل الألعاب الكلاسيكية قريباً');
    };

    // Deposit function
    window.deposit = function() {
      showNotification('سيتم فتح صفحة الإيداع قريباً');
    };
  </script>
